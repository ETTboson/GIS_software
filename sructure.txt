│   ├── analysismodule.h/cpp           # 旧版分析模块（保留）

├── core/                              # 基础核心层，负责 AI 编排与基础设施

│   ├── dataservice.h/cpp              # 数据加载入口，分发 CSV/简单栅格/GIS 文件解析

└── ui/                                # 表现层，只负责交互与结果展示
    ├── mainwindow.h/cpp               # 主窗口，组织数据加载、分析面板与结果展示
    ├── docks/

    ├── style/

    └── widgets/

        └── mapcanvasmanager.h/cpp     # 多画布管理器
