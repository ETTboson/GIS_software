rules.txt                              # 项目代码规范、命名规则与分层要求

resources/

├── model/                             # 纯数据结构层，不含业务逻辑

│   │   ├── chatmessage.h              # AI 对话消息 DTO
│   │   ├── analysisresult.h           # 分析结果 DTO
│   │   ├── numericdataset.h           # 数值型数据集 DTO（CSV/简单栅格分析输入）
│   │   ├── visualizationdata.h        # 图表展示 DTO，承载标题、坐标轴与系列列表

│   ├── enums/

│   │   └── visualizationcharttype.h   # 可视化图表类型枚举
│
├── core/                              # 基础核心层，负责 AI 编排、工作流与运行时集成
│   ├── ai/
│   │   ├── aimanager.h/cpp            # AI 对话总协调层，负责普通链/分析状态更新链切换与 tool_call 兼容
│   │   ├── conversationcontext.h/cpp  # 对话上下文、历史压缩与 ET.md 记忆注入管理
│   │   ├── ollamaclient.h/cpp         # 与 Ollama 的 HTTP 流式通信层，支持 NDJSON 半包缓冲与中断收尾
│   │   ├── toolcalldispatcher.h/cpp   # AI 工具定义、参数校验与宿主桥接分发层
│   │   └── memory/

│   │       └── memoryfilemanager.h/cpp    # ET.md / memory/*.md 记忆文件管理
│   ├── workflow/
│   │   └── analysisworkflowcoordinator.h/cpp # 分析工作流状态机，负责分析类型识别、缺参收集与内部状态更新
│   ├── integration/
│   │   └── qgsappinitializer.h/cpp    # QGIS 应用初始化/释放管理
│   └── interfaces/
│       └── iaitoolhost.h              # AI 宿主桥接接口，向 AI 暴露分析上下文与分析执行能力
│
├── service/                           # 业务逻辑层，负责数据读取与分析算法
│   ├── data/
│   │   ├── dataservice.h/cpp          # 数据加载入口，分发 CSV/简单栅格/GIS 文件解析
│   │   └── numericdatareader.h/cpp    # 数值数据读取器，负责 CSV 与简单栅格文本解析
│   └── analysis/
│       └── analysisservice.h/cpp      # 数据分析引擎，提供基础统计、频率统计、邻域分析与 AI 工具桥接
│
└── ui/                                # 表现层，只负责交互与结果展示
    ├── mainwindow.h/cpp               # 主窗口，组织数据加载、分析面板与结果展示
    ├── docks/

    ├── components/

    ├── map/

    └── visualization/
        ├── visualizationdockwidget.h/cpp # 可视化结果 Dock，承载占位页、图表页与详情区

        └── charts/
            ├── chartwidget.h/cpp      # 图表基类，统一处理 DTO 绑定、悬停 tooltip 与点击选中
            ├── barchartwidget.h/cpp   # 柱状图控件，负责基础统计与频率统计展示
            └── linechartwidget.h/cpp  # 折线图控件，负责邻域分析行趋势展示
